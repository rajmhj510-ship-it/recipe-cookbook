document.addEventListener("DOMContentLoaded", async () => { const id = Number(new URLSearchParams(location.search).get("id")); const content = document.getElementById("content"); const hero = document.getElementById("hero"); try { const [a, b] = await Promise.all([ fetch("./data/arabic-recipes.json"), fetch("./data/chinese-recipes.json") ]); const data = [...await a.json(), ...await b.json()]; const recipe = data.find(r => r.id === id); if(!recipe){ content.innerHTML = "Recipe not found"; return; } hero.style.backgroundImage = `url('${recipe.image}')`; content.innerHTML = `

${recipe.title}

${recipe.time} ${recipe.difficulty} ${recipe.category}

Ingredients

${recipe.ingredients.map(g => `

${g.group}

    ${g.items.map(i => `
  • ${i}
    • `).join("")}
`).join("")}

Instructions

${recipe.instructions.map(g => `

${g.group}

${g.steps.map(s => `
${s}
`).join("")}
`).join("")} ${recipe.tips ? `

Chef Tips

    ${recipe.tips.map(t => `
  • ${t}
    • `).join("")}
` : ""} `; } catch (e) { console.error(e); content.innerHTML = "Failed to load recipe ❌"; } });
